import java.util.stream.Stream;
import java.io.*;

public class Pattern4 {
  public static void main(String[] args) throws IOException {
    int l = Integer.parseInt(new BufferedReader(new InputStreamReader(System.in)).readLine());
    if(l % 2 == 0)
      System.out.print("Invalid line number");
    else {
      int half = l / 2;
      for(int i = 1; i <= half + 1; i++) {
        Stream.generate(() -> " ").limit(half - i + 1).forEach(System.out::print);
        Stream.generate(() -> "* ").limit(i).forEach(System.out::print);
        System.out.println("");
      }
      for(int i = half; i >= 1; i--) {
        Stream.generate(() -> " ").limit(half - i).forEach(System.out::print);
        Stream.generate(() -> " *").limit(i).forEach(System.out::print);
        System.out.println("");
      }
    }
  }
}
